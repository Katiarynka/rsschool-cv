        let pairsOfBrackets = new Object();

            pairsOfBrackets[bracketsConfig[i][1]] = bracketsConfig[i][0];

        for (let i = 0; i < str.length; i++) {

                if (pairsOfBrackets[str[i]] === str[i]) {
                    if (stack.length !== 0 && stack[stack.length - 1] === str[i]) { stack.pop() }

            }
        }
